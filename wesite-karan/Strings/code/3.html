3. Enter the String from the user and convert in Sentence Case

        #include < stdio.h >

        int main() {
            int i;
            char str[100];
            printf("Enter a string: ");
            gets(str);
            if(str[0] >= 'a' && str[0] <= 'z') {
                    str[0] -= 32 ;
            }
                
            for(i = 1; str[i] != '\0'; i++) {
                if(str[i] >= 'A' && str[i] <= 'Z') {
                    str[i] += 32 ;
                }
                        
            }
            printf("Sentence Case: %s",str);
                
            }

        
Output

        Enter a string: HALL OF FAME
        Sentence Case: Hall of fame
