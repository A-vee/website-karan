17. Enter two Strings from the user and count the number of vowels.

        #include < stdio.h>

        int main() {
            int i,count=0;
            char str[100];
            printf("Enter a string: ");
            gets(str);
        
            for(i = 0; str[i] != '\0'; i++) {
            if(str[i] == 'a' || 
               str[i] == 'e' ||
               str[i] == 'i' ||
               str[i] == 'o' ||
               str[i] == 'u') {
                count++;
            }
            }
        
            printf("Total number of vowels %d",count);
        
        }
        
Output

        Enter a string: Crown and Jewells
        Total number of vowels 4
