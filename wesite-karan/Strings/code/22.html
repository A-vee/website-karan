22. Enter the String from the user and sort the string in ascending order.

        #include < stdio.h>

        int main() {
            int i,j,lenght=0;
            char str[100];
            printf("Enter a string");
            gets(str);
        
            for (i = 0; str[i] != '\0'; i++) {
                lenght++;
            }
        
            for(j=0;j< lenght-1;j++) {
                for(i=j+1;i< lenght;i++) 
                {
                if(str[i]< str[j])
                {
                   char temp = str[i];
                   str[i] = str[j];
                   str[j] = temp;
                }
                } 
            }
        
            printf("Ascending Order: %s ",str);   
        }
