7. Enter an array from the user and square all the values of array.

        #include < stdio.h>
        int main() {
            int i,arr[5];
            printf("Enter the values: ");
            for(i=0;i< 5;i++) {
                scanf("%d",&arr[i]);
            }
            printf("Square Numbers: ");
            for(i=0;i< 5;i++) {
                printf("%d ",arr[i]*arr[i]);
            }
            
        }
        
        
Output

        Enter the values: 1 28 76 10 8
        Square Numbers: 1 784 5776 100 64
