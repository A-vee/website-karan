5. Enter the String from the user and convert in ToggleCase

        #include < stdio.h >
        int main() {
            int i;
            char str[100];
            printf("Enter a string: ");
            gets(str);
            for(i = 0; str[i] != '\0'; i++) {
                if(str[i] >= 'A' && str[i] <= 'Z') {
                    str[i] += 32 ;
                }
                else if(str[i] >= 'a' && str[i] <= 'z') {
                    str[i] -= 32 ;
                }
            }
            printf("ToggleCase: %s",str);
        
        }
       
        
Output

        Enter a string: gaMe oF ThrOnEs
        ToggleCase: GAmE Of tHRoNeS
