11. Enter an array from the user and find the duplicate values.

        #include < stdio.h>
        int main() {
            int n,count=0;
            printf("Enter the size of array: ");
            scanf("%d",&n);
            int arr[n];
        
            printf("Enter values: ");
            for(int i=0;i< n;i++) {
                scanf("%d",&arr[i]);
            }
        
            printf("Duplicate Values: ");
            for(int i=0;i< n-1;i++) {
                for(int j=i+1;j< n;j++) {
                    if(arr[i] == arr[j]) {
                        printf("%d  ",arr[i]);
                        break;
                    }
                }
            }  
            printf("\n");
        }
