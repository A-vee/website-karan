6. Enter the string from the user and find the length of the String.

        #include < stdio.h>
        int main() {
            int i,count=0;
            char str[100];
            printf("Enter a string: ");
            gets(str);
        
            for(i = 0; str[i] != '\0'; i++) {
                count++;
            }
            printf("Length of String: %d",count);
        
        }

        
Output

        Enter a string: Attack on Titan
        Length of String: 15
