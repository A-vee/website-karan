8. Enter an array from the user and find maximum and minimum value.
    
        #include < stdio.h>
        int main() {
            int i,arr[5];
            printf("Enter the values: ");
            for(i=0;i< 5;i++) {
                scanf("%d",&arr[i]);
            }
            int max=arr[0];
            int min=arr[0];
        
            for(i=1;i< 5;i++) {
            if(arr[i]>max) {
                max = arr[i];
            }
            if(arr[i]< min) {
                min = arr[i];
            }
            }
           printf("Maximum value: %d",max);
           printf("\nMinimum value: %d",min);
        
        }    
        
        
Output
