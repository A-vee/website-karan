11. Enter the String from the user and check its Pallindrome or not.

        #include < stdio.h>

        int main() {
            int i,j=0,l=1;
            char str1[100];
            printf("Enter a string: ");
            gets(str1);
            
            for(i=0;str1[i]!='\0';i++);
        
            i--;
        
            for(j=0;j< i;j++) {
            if(str1[j] != str1[i-j]) {
                l=0;
                break;
            }
            }
        
            if(l==1) {
                printf("Yes, the string is Pallindrome");
            }
            else 
            printf("No, the string is not Pallindrome");
        
        }
