12. Enter an array from the user and remove the duplicate values and fill with zero.

        #include < stdio.h>
        int main() {
            int n,pass=0;
            printf("Enter the size of array: ");
            scanf("%d",&n);
            int arr[n];
        
            printf("Enter values: ");
            for(int i=0;i< n;i++) {
                scanf("%d",&arr[i]);
            }
        
        
            for(int i=0;i< n;i++) {
                for(int j=i+1;j< n;j++) {
                    if(arr[i] == arr[j]) {
                        arr[j] = 0;
                    }
                }
            }  
        
            printf("Modified Array: "); 
            for(int i=0;i< n;i++) {
                printf("%d ",arr[i]);
            }   
        }

        
Output

        Enter values: 1 2 2 6 3
