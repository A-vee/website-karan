9. Enter an array from the user and find the k-th smallest value.
  
        #include < stdio.h>
        int main() {
            int n,k,temp;
            printf("Enter the size of array: ");
            scanf("%d",&n);
            int arr[n];
        
            printf("Enter values: ");
            for(int i=0;i< n;i++) {
                scanf("%d",&arr[i]);
            }
        
            printf("Enter the value of k: ");
            scanf("%d",&k);
            
            for(int j=0;j< n-1;j++) {
               for(int i=0;i< n-j-1;i++) {
               if(arr[i] > arr [i+1]) {
                temp = arr[i];
                arr[i] = arr[i+1];
                arr[i+1] = temp;
               }
              } 
            }
            
            printf("K-th smallest: %d",arr[k-1]);
        }
